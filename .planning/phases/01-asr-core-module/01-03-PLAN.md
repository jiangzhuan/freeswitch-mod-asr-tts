       - `void worker_pool_destroy(worker_pool_t **pool)` — stop threads and cleanup

    3. Worker thread function:
       ```c
       static void* worker_thread_func(switch_thread_t *thread, void *data) {
           worker_pool_t *pool = (worker_pool_t *)data;
           while (pool->running) {
               worker_task_t *task = NULL;
               switch_mutex_lock(pool->queue_mutex);
               while (switch_queue_size(pool->task_queue) == 0 && pool->running) {
                   switch_thread_cond_wait(pool->queue_cond, pool->queue_mutex);
               }
               if (!pool->running) break;
               switch_queue_pop(pool->task_queue, (void**)&task);
               switch_mutex_unlock(pool->queue_mutex);
               if (task) {
                   task->func(task->data);
                   free(task);  // Task freed after execution
               }
           }
           return NULL;
       }
       ```
    
    4. Use FreeSWITCH primitives:
       - `switch_queue_create()` for task queue
       - `switch_thread_create()` for worker threads
       - `switch_thread_cond_create()` for condition variable

       static float calculate_energy_db(int16_t *audio, size_t len) {
           double sum = 0;
           for (size_t i = 0; i < len; i++) {
               sum += audio[i] * audio[i];
           }
           double rms = sqrt(sum / len);
           double db = 20 * log10(rms / 32768.0);  // Relative to max int16

       }

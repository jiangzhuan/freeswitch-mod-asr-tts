       ```c
           switch_core_session_t *session;
           asr_ring_buffer_t *buffer;      // Audio buffer
           asr_resampler_t *resampler;     // 8kHz -> 16kHz resampler
           switch_media_bug_t *bug;        // Media bug handle
           switch_memory_pool_t *pool;     // Session memory pool

       ```
    
    2. Media bug callback:
       ```c
       static switch_bool_t asr_media_bug_callback(switch_media_bug_t *bug, void *user_data, 
                                                    switch_abc_type_t type)
       {
           asr_session_ctx_t *ctx = (asr_session_ctx_t *)user_data;
           
           switch (type) {
               case SWITCH_ABC_TYPE_INIT:
                   ctx->active = SWITCH_TRUE;
                   switch_log_printf(SWITCH_CHANNEL_SESSION_LOG(ctx->session), SWITCH_LOG_INFO, "ASR session initialized");
                   break;
                   
               case SWITCH_ABC_TYPE_READ_REPLACE:
                   if (ctx->active) {
                       switch_frame_t *frame = switch_core_media_bug_get_read_replace_frame(bug);
                       // Resample and buffer (NON-BLOCKING per Pitfall 4)
                       int16_t resampled[640];  // Max output for 20ms at 16kHz
                       size_t resampled_len = sizeof(resampled);
                       resampler_process(ctx->resampler, frame->data, frame->datalen/2, resampled, &resampled_len);
                       buffer_write(ctx->buffer, resampled, resampled_len);
                       ctx->frame_count++;
                   }
                   break;
                   
               case SWITCH_ABC_TYPE_CLOSE:
                   ctx->active = SWITCH_FALSE;
                   switch_log_printf(SWITCH_CHANNEL_SESSION_LOG(ctx->session), SWITCH_LOG_INFO, "ASR session closed, frames: %u", ctx->frame_count);
                   break;
           }
           return SWITCH_TRUE;
       }
       ```
    
    3. Public functions:
       - `switch_status_t media_bug_attach(asr_session_ctx_t **ctx, switch_core_session_t *session)` — attach media bug to session
         - Create session pool with `switch_core_new_memory_pool(&pool)`
         - Create buffer and resampler
         - Call `switch_core_media_bug_add(session, "asr", NULL, asr_media_bug_callback, ctx, 0, SMBF_READ_STREAM_REPLACE, &ctx->bug)`
       - `void media_bug_detach(asr_session_ctx_t **ctx)` — detach and cleanup
         - Remove bug with `switch_core_media_bug_remove(session, &bug)`
         - Destroy buffer and resampler
         - Destroy memory pool with `switch_core_destroy_memory_pool(&pool)`
    
    4. Critical: Non-blocking (per Pitfall 4):
       - Callback MUST NOT make network calls
       - Callback MUST NOT wait on locks for extended time
       - All heavy processing deferred to worker threads (will be added in Plan 03)
